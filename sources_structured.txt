   280	    private Object convertToDbValue(Field f, Object raw) throws IllegalAccessException {
   281	        if (raw == null) return null;
   282	
   283	        // FK : ne prendre que l'ID de l'entité liée
   284	        if (f.isAnnotationPresent(ManyToOne.class) || isEntity(raw.getClass())) {

   287	        // Enum

   289	            Enumerated enumAnn = f.getAnnotation(Enumerated.class);
   290	            if (enumAnn != null && enumAnn.value() == Enumerated.EnumType.STRING) {
   291	                return ((Enum<?>) raw).name();
   292	            } else {
   293	                return ((Enum<?>) raw).ordinal();
   294	            }
   295	        }
   296	        // Champ JSON (chaîne sérialisée, sera traitée plus tard pour le cast PGobject)
   297	        if (raw instanceof String && f.getName().equals("infosSupplementairesJson")) {
   298	            return raw;
   299	        }
   300	        return raw;

   303	    private void setFieldFromDbValue(T entity, Field f, Object dbValue, int depth, Map<Pair, Object> cache) throws Exception {
   304	        if (f.isAnnotationPresent(ManyToOne.class) || isEntity(f.getType())) {
   305	            if (dbValue != null) {
   306	                Class<?> targetClass = f.getType();
   307	                GenericDao<?> targetDao = new GenericDao<>(targetClass);
   308	                Object related = targetDao.findById(dbValue, depth, cache);
   309	                f.set(entity, related);
   310	            }
   311	        } else if (f.getType().isEnum()) {
   312	            Enumerated enumAnn = f.getAnnotation(Enumerated.class);
   313	            if (enumAnn != null && enumAnn.value() == Enumerated.EnumType.STRING) {
   314	                f.set(entity, Enum.valueOf((Class<Enum>) f.getType(), (String) dbValue));
   315	            } else {
   316	                int ordinal = ((Number) dbValue).intValue();
   317	                Object[] constants = f.getType().getEnumConstants();
   318	                f.set(entity, constants[ordinal]);
   319	            }
   320	        } else {
   321	            // Conversion basique
   322	            if (f.getType() == int.class || f.getType() == Integer.class) {
   323	                f.set(entity, ((Number) dbValue).intValue());
   324	            } else if (f.getType() == String.class) {
   325	                f.set(entity, dbValue.toString());
   326	            } else {
   327	                f.set(entity, dbValue);
   328	            }
   329	        }

   333	                                                 int depth, Map<Pair, Object> cache) throws Exception {
   334	        List<Object> items = new ArrayList<>();
   335	        String sql = "SELECT t.* FROM " + ann.joinTable() + " jt " +
   336	                     "JOIN " + getTableNameFromJoinTable(ann.inverseJoinColumn()) + " t " +
   337	                     "ON jt." + ann.inverseJoinColumn() + " = t.id " +
   338	                     "WHERE jt." + ann.joinColumn() + " = ?";
   339	        try (PreparedStatement ps = conn.prepareStatement(sql)) {
   340	            ps.setObject(1, ownerId);
   341	            ResultSet rs = ps.executeQuery();
   342	            Class<?> targetClass = getTargetClassOfCollectionField(ann);
   343	            GenericDao<?> targetDao = new GenericDao<>(targetClass);
   344	            while (rs.next()) {
   345	                Object itemId = rs.getObject("id");
   346	                Object item = targetDao.findById(itemId, depth, cache);
   347	                if (item != null) items.add(item);
   348	            }
   349	        }
   350	        return items;

     7	import annotation.Id;
     8	import annotation.IgnoredField;

    10	/**
    11	 * Génère dynamiquement des requêtes SQL à partir d'une classe Java.
    12	 * Respecte @Id et @IgnoredField.
    13	 *
    14	 * Exemple pour Personne { @Id String id; String nom; }
    15	 *   INSERT INTO personne(id,nom) VALUES(?,?)
    16	 *   UPDATE personne SET nom=? WHERE id=?
    17	 *   DELETE FROM personne WHERE id=?
    18	 *   SELECT * FROM personne
    19	 */
    20	public class SqlGenerator {
    21	
    22	    // -------------------------------------------------------
    23	    // INSERT INTO table(col1,col2) VALUES(?,?)
    24	    // -------------------------------------------------------
    25	    public static String generateInsert(Class<?> clazz) {
    26	        StringBuilder columns = new StringBuilder();
    27	        StringBuilder values  = new StringBuilder();
    28	
    29	        List<Field> fields = getActiveFields(clazz);
    30	
    31	        for (int i = 0; i < fields.size(); i++) {
    32	            columns.append(fields.get(i).getName());
    33	            values.append("?");
    34	            if (i < fields.size() - 1) {
    35	                columns.append(",");
    36	                values.append(",");
    37	            }
    38	        }
    39	
    40	        return "INSERT INTO " + tableName(clazz)
    41	                + "(" + columns + ") VALUES(" + values + ")";
    42	    }
    43	
    44	    // -------------------------------------------------------
    45	    // UPDATE table SET col1=?,col2=? WHERE idCol=?
    46	    // -------------------------------------------------------
    47	    public static String generateUpdate(Class<?> clazz) {
    48	        String idCol   = findIdFieldName(clazz);
    49	        StringBuilder set = new StringBuilder();
    50	
    51	        for (Field f : getActiveFields(clazz)) {
    52	            if (f.getName().equals(idCol)) continue; // l'ID va dans WHERE
    53	            set.append(f.getName()).append("=?,");
    54	        }
    55	        // Supprimer la dernière virgule
    56	        if (set.length() > 0) set.setLength(set.length() - 1);
    57	
    58	        return "UPDATE " + tableName(clazz)
    59	                + " SET " + set
    60	                + " WHERE " + idCol + "=?";
    61	    }
    62	
    63	    // -------------------------------------------------------
    64	    // DELETE FROM table WHERE idCol=?
    65	    // -------------------------------------------------------
    66	    public static String generateDelete(Class<?> clazz) {
    67	        return "DELETE FROM " + tableName(clazz)
    68	                + " WHERE " + findIdFieldName(clazz) + "=?";
    69	    }
    70	
    71	    // -------------------------------------------------------
    72	    // SELECT * FROM table
    73	    // -------------------------------------------------------
    74	    public static String generateSelectAll(Class<?> clazz) {
    75	        return "SELECT * FROM " + tableName(clazz);
    76	    }
    77	
    78	    // -------------------------------------------------------
    79	    // SELECT * FROM table WHERE idCol=?
    80	    // -------------------------------------------------------
    81	    public static String generateSelectById(Class<?> clazz) {
    82	        return "SELECT * FROM " + tableName(clazz)
    83	                + " WHERE " + findIdFieldName(clazz) + "=?";
    84	    }
    85	
    86	    // -------------------------------------------------------
    87	    // Liste des noms de colonnes (pour JTable header, etc.)
    88	    // -------------------------------------------------------
    89	    public static List<String> getColumnNames(Class<?> clazz) {
    90	        List<String> names = new ArrayList<>();
    91	        for (Field f : getActiveFields(clazz)) {
    92	            names.add(f.getName());
    93	        }
    94	        return names;
    95	    }
    96	
    97	    // -------------------------------------------------------
    98	    // Helpers privés
    99	    // -------------------------------------------------------
   100	
   101	    /**
   102	     * Retourne le nom de la table :
   103	     * @Table(name="…") ou nom simple de la classe en minuscules.
   104	     */
   105	    public static String tableName(Class<?> clazz) {
   106	        if (clazz.isAnnotationPresent(annotation.Table.class)) {
   107	            String n = clazz.getAnnotation(annotation.Table.class).name();
   108	            return n.isEmpty() ? clazz.getSimpleName().toLowerCase() : n;
   109	        }
   110	        return clazz.getSimpleName().toLowerCase();
   111	    }
   112	
   113	    /**
   114	     * Retourne les champs actifs : ni @IgnoredField.
   115	     */
   116	    public static List<Field> getActiveFields(Class<?> clazz) {
   117	        List<Field> result = new ArrayList<>();
   118	        for (Field f : clazz.getDeclaredFields()) {
   119	            if (!f.isAnnotationPresent(IgnoredField.class)) {
   120	                result.add(f);
   121	            }
   122	        }
   123	        return result;
   124	    }
   125	
   126	    /**
   127	     * Trouve le nom du champ @Id.
   128	     * Si aucun @Id n'est présent, utilise "id" par convention.
   129	     */
   130	    public static String findIdFieldName(Class<?> clazz) {
   131	        for (Field f : clazz.getDeclaredFields()) {
   132	            if (f.isAnnotationPresent(Id.class)) return f.getName();
   133	        }
   134	        return "id"; // convention par défaut
   135	    }
   136	}

    32	        info.setClassName(clazz.getSimpleName());

    34	        // Nom de table : @Table(name=…) ou nom simple de la classe

    52	     * Respecte @Id et @IgnoredField.

    54	     *
    55	     * CORRECTION : utilise List.class.isAssignableFrom(type)
    56	     *              au lieu de type == List.class

    62

    76	
    77	            // --- @Id ---
    78	            if (field.isAnnotationPresent(Id.class)) {
    79	                info.setIdField(true);
    80	            }
    81	
    82	            Class<?> type = field.getType();
    83	
    84	            // ---- ENUM ----
    85	            if (type.isEnum()) {
    86	                info.setEnumType(true);
    87	                info.setTypeField(TypeField.ENUM);
    88	            }
    89	
    90	            // ---- LIST<T> ----

    94	                info.setTypeField(TypeField.LIST);

   110

   121	            }

    20

    22	                case SAVE -> handleSave(event, repo);
    23	                case UPDATE -> handleUpdate(event, repo);
    24	                case DELETE -> handleDelete(event, repo);

    38	        }

    40	
    41	    private void handleUpdate(FormEvent event, CrudRepository<Object> repo) throws Exception {
    42	        Object entity = event.getSourceEntity();
    43	        if (entity == null) {
    44	            throw new IllegalStateException("Pas d'entité source pour UPDATE");
    45	        }
    46	        applyFormDataToEntity(event, entity);
    47	        repo.update(entity);
    48	        if (successCallback != null) {
    49	            successCallback.onSuccess(event, entity);

    51	    }
    52	
    53	    private void handleDelete(FormEvent event, CrudRepository<Object> repo) throws Exception {
    54	        Object entity = event.getSourceEntity();
    55	        if (entity == null) {
    56	            throw new IllegalStateException("Pas d'entité source pour DELETE");
    57	        }

    59	        if (successCallback != null) {
    60	            successCallback.onSuccess(event, entity);
    61	        }
    62	    }

    64	    private Object createEntityFromFormData(FormEvent event) throws Exception {
    65	        Object entity = event.getEntityClass().getDeclaredConstructor().newInstance();

    71	        for (var entry : event.getFormData().entrySet()) {
    72	            String fieldName = entry.getKey();
    73	            Object fieldValue = entry.getValue();
    74	            try {
    75	                java.lang.reflect.Field field = 
    76	                    event.getEntityClass().getDeclaredField(fieldName);
    77	                field.setAccessible(true);
    78	                field.set(entity, fieldValue);
    79	            } catch (NoSuchFieldException e) {
    80	                // Champ ignoré
    81	            }
    82	        }
    83	    }

    88	                          + exception.getMessage());
